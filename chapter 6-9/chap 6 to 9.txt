var a = 10;

++a;
document.write("<br>The value of ++a is "+a);

document.write("<br> Now the value of a is : "+a);

document.write("<br><br><br>The value of a++ is : "+(a++));
document.write("<br>Now the value of a is : "+a);

--a;
document.write("<br><br><br>The value of --a is : "+a);

document.write("<br>Now the value of a is : "+a);

document.write("<br><br><br>The value of a-- is : "+(a--));
document.write("<br>The value of a is : "+a);
